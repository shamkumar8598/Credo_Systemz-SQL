-- Q14. Find total sales per city.

SELECT city,sum(total_amount)
From customer as C 
LEFT join orders as O
ON C.Customer_id=O.customer_id

FROM orders

🧩 7. HAVING Clause

-- Q16. Find customers whose total purchase value is greater than ₹3000.

SELECT customer_id, SUM(total_amount) AS Total_purchase
FROM orders
GROUP BY customer_id
HAVING Total_purchase>3000;

-- Q17. Find cities with total sales greater than ₹5000.

SELECT city,sum(total_amount) AS Total_salse
From customer as C 
LEFT join orders as O
ON C.Customer_id=O.customer_id
GROUP BY C.City
HAVING Total_salse>5000

SELECT C.name,C.City

-- Q20. Join orders and products to display order_id, product name, price (assume order_products table links them)

SELECT O.order_id, P.product_name, P.price
FROM Orders AS O
JOIN order_products AS OP
ON O.order_id = OP.order_id
JOIN Products AS P
 ON OP.product_id = P.product_id;


🧩 9. Aggregate + Join (Analyst Scenario)


-- Q21. Show each city’s total revenue and number of unique customers.

SELECT c.city,SUM(o.total_amount) AS total_revenue,
COUNT(DISTINCT c.customer_id) AS unique_customers
FROM customer c
JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.city;

-- Q22. Find the top 3 customers by total order value.

SELECT c.name, SUM(o.total_amount) AS total
FROM customer c
JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.name
ORDER BY total DESC
LIMIT 3;

🧩 10. Window Functions (Analytical Queries)

-- Q23. Rank customers by total purchase amount. 

SELECT c.name,
SUM(o.total_amount) AS total,
RANK() OVER (ORDER BY SUM(o.total_amount) DESC) AS rank
FROM customer c
JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.name
ORDER BY rank ;

-- Q24. Calculate a running total of sales ordered by date. 
SELECT order_date,SUM(total_amount) OVER (ORDER BY order_date) AS running_total
FROM orders;


-- Q25. Find the monthly average order amount and show each order’s deviation from that average.
SELECT order_id, order_date, total_amount,
 AVG(total_amount) OVER (
        PARTITION BY YEAR(order_date), MONTH(order_date)) AS monthly_avg,
total_amount - AVG(total_amount) OVER (
        PARTITION BY YEAR(order_date), MONTH(order_date)) AS deviation
FROM Orders;
